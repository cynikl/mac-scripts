date >> /path/to/setregproptool/srptlog.txt

if [ $? == 0 ] ; then
	echo "Already Set" >> /path/to/setregproptool/srptlog.txt
elif [ $? == 1 ] ; then
	echo "Setting Now" >> /path/to/setregproptool/srptlog.txt
